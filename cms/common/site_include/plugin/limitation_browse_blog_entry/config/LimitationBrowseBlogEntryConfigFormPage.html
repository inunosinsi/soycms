<!-- b_block:id="entry" -->
<h2 cms:id="title_plain">タイトル</h2>

<form cms:id="browse_password_form">
	<input type="text" cms:id="browse_password_input">
	<input type="submit" value="送信">
</form>
<!-- /cms:id="not_allow_browse_entry" -->
